```cpp
void WaveformDisplayComponent::drawThumbnail(Graphics& g)
{
    // Get the audio data from the DJAudioPlayer
    AudioData data = djaudioplayer.getAudioData();

    // Create a new Rectangle to hold our waveform thumbnail
    Rectangle rectangle;

    // Draw the waveform on the screen
    for (int i = 0; i < data.size(); i++)
    {
        if (data[i] > 0)
        {
            g.drawRectangle(rectangle, Color::black);
        }
        else
        {
            g.drawRectangle(rectangle, Color::white);
        }
        rectangle.move(1, 0);
    }

    // Draw the thumbnail on the screen
    g.drawThumbnail(rectangle);
}
```

Here is a step-by-step guide to setting up the OrderBook class:

**Step 1: Create a new class**

Create a new C++ file called `OrderBook.h` and add the following code:
```cpp
#ifndef ORDERBOOK_H
#define ORDERBOOK_H

#include <map>
#include <vector>

class Order;
class Account;

class OrderBook {
public:
    // Constructor
    OrderBook();

    // Destructor
    ~OrderBook();

    // Get known products
    std::map<std::string, double> getKnownProducts();

    // Get orders by product and time
    std::map<std::string, std::vector<Order>> getOrdersByProductAndTime(const std::string& product);

    // Add an order to the book
    void addOrder(Order* order);

private:
    std::map<std::string, double> products_;
    std::map<std::string, std::vector<Order>> orders_;
};

#endif  // ORDERBOOK_H
```

**Step 4: Implement getKnownProducts**

Implement the `getKnownProducts` function to return a map of known products:
```cpp
std::map<std::string, double> OrderBook::getKnownProducts() {
    std::map<std::string, double> result;
    for (const auto& product : products_) {
        result[product.first] = product.second;
    }
    return result;
}
```
**Step 5: Implement getOrdersByProductAndTime**

Implement the `getOrdersByProductAndTime` function to return a map of orders by product and time:
```cpp
std::map<std::string, std::vector<Order>> OrderBook::getOrdersByProductAndTime(const std::string& product) {
    std::map<std::string, std::vector<Order>> result;
    for (const auto& order : orders_) {
        if (order.first == product && !order.second.empty()) {
            result[product] = order.second;
        }
    }
    return result;
}
```
**Step 6: Implement addOrder**

Implement the `addOrder` function to add an order to the book:
```cpp
void OrderBook::addOrder(Order* order) {
    std::string product = order->getProduct();
    if (products_.find(product) == products_.end()) {
        products_[product] = 0.0;
    }
    orders_[product].push_back(order);
}
```

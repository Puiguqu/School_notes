class OrderBook {
public:
    // Constructor
    OrderBook() {}
    
    // Method to get the total cash available for bids and asks
    float getCash() {
        float totalCash = 0;
        for (Order order : orders_) {
            if (order.side == 1) {
                totalCash += order.amount();
            } else if (order.side == -1) {
                totalCash -= order.amount();
            }
        }
        return totalCash;
    }

private:
    // List of orders
    vector<Order> orders_;
};

The `OrderBook` class represents a book of orders, where each order has a symbol, side, and amount. The `getCash` method calculates the total cash available in the book by summing up the amounts of all buy orders and subtracting the amounts of all sell orders.
